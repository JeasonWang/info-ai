# 信息达人 MVP 快速落地 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在现有 `info_aggregation` 和 `info-max` 基础上，尽快交付一个可运行的事件流 MVP，支持首页事件流、分类切换、事件详情、一句话看懂和时间线展示。

**Architecture:** 后端继续以 Python/FastAPI/SQLAlchemy 为主，保留现有 `info` 作为内容层，新增 `event` 相关表和规则版事件重建服务；前端以 Vue3 为主，新增事件流与事件详情视图，直接消费 `/api/events`、`/api/events/{id}` 和 `/api/event-categories`。MVP 不强制引入重型 MQ，先通过轻量异步重建和显式重建入口跑通闭环。

**Tech Stack:** Python、FastAPI、SQLAlchemy、PyMySQL、SQLite/MySQL、Vue3、TypeScript、Vite、Vitest、MySQL、Redis（后续接入）

import logging
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker, scoped_session

from config import SQLALCHEMY_DATABASE_URL
from .models import Base

logger = logging.getLogger(__name__)

engine = None
SessionFactory = None
Session = None


def configure_engine(database_url: str):
    global engine, SessionFactory, Session
    engine = create_engine(
        database_url,
        echo=False,
        pool_pre_ping=True,
        connect_args={"check_same_thread": False} if "sqlite" in database_url else {},
    )
    SessionFactory = sessionmaker(bind=engine, autocommit=False, autoflush=False)
    Session = scoped_session(SessionFactory)


configure_engine(SQLALCHEMY_DATABASE_URL)

# info_aggregation/database/models.py
class Event(Base):
    __tablename__ = "event"

    id = Column(Integer, primary_key=True, autoincrement=True)
    title = Column(String(200), nullable=False)
    one_line_summary = Column(String(255), default="")
    primary_category_id = Column(Integer, ForeignKey("category.id"), nullable=False)
    status = Column(String(20), default="active")
    heat_score = Column(Integer, default=0)
    freshness_score = Column(Integer, default=0)
    composite_score = Column(Integer, default=0)
    source_count = Column(Integer, default=0)
    started_at = Column(DateTime)
    last_updated_at = Column(DateTime)
    created_at = Column(DateTime, default=datetime.now)
    updated_at = Column(DateTime, default=datetime.now, onupdate=datetime.now)

    category = relationship("Category", lazy="joined")


class EventItemLink(Base):
    __tablename__ = "event_item_link"

    id = Column(Integer, primary_key=True, autoincrement=True)
    event_id = Column(Integer, ForeignKey("event.id"), nullable=False)
    item_id = Column(Integer, ForeignKey("info.id"), nullable=False)
    role = Column(String(20), default="media")
    is_primary = Column(Integer, default=0)
    weight = Column(Integer, default=0)
    created_at = Column(DateTime, default=datetime.now)

    __table_args__ = (
        UniqueConstraint("event_id", "item_id", name="uq_event_item"),
        Index("idx_event_item_item_id", "item_id"),
    )


class EventTimelineEntry(Base):
    __tablename__ = "event_timeline_entry"

    id = Column(Integer, primary_key=True, autoincrement=True)
    event_id = Column(Integer, ForeignKey("event.id"), nullable=False)
    occurred_at = Column(DateTime, nullable=False)
    summary = Column(String(255), nullable=False)
    source_item_id = Column(Integer, ForeignKey("info.id"), nullable=False)
    confidence = Column(Float, default=0.0)
    display_order = Column(Integer, default=0)
    created_at = Column(DateTime, default=datetime.now)

    __table_args__ = (Index("idx_event_timeline_event_time", "event_id", "occurred_at"),)


class EventSummarySnapshot(Base):
    __tablename__ = "event_summary_snapshot"

    id = Column(Integer, primary_key=True, autoincrement=True)
    event_id = Column(Integer, ForeignKey("event.id"), nullable=False)
    summary_type = Column(String(30), nullable=False)
    content = Column(Text, default="")
    version = Column(Integer, default=1)
    created_at = Column(DateTime, default=datetime.now)

    __table_args__ = (Index("idx_event_summary_lookup", "event_id", "summary_type", "version"),)

# info_aggregation/services/event_builder.py
import re
from collections import defaultdict

from database import Event, EventItemLink, EventSummarySnapshot, EventTimelineEntry, Info


def build_event_key(item: Info) -> tuple[int, str]:
    anchor = (item.core_entity or "").strip().lower()
    if not anchor:
        tokens = re.findall(r"[\u4e00-\u9fffA-Za-z0-9]+", item.title.lower())
        anchor = "".join(tokens[:4])
    return item.category_id, anchor or item.title[:12].lower()


def build_one_line(items: list[Info]) -> str:
    latest = items[0]
    return f"{latest.core_entity or latest.title[:12]} 相关讨论正在升温，当前共聚合 {len(items)} 条来源内容。"


def build_why_it_matters(items: list[Info]) -> str:
    latest = items[0]
    return f"这条事件目前在 {len({item.channel_id for item in items})} 个来源出现，说明它已经超出单个平台讨论范围。"


def rebuild_events(session, limit: int = 200):
    recent_items = (
        session.query(Info)
        .filter(Info.is_deleted == 0)
        .order_by(Info.event_time.desc(), Info.created_at.desc())
        .limit(limit)
        .all()
    )
    groups = defaultdict(list)
    for item in recent_items:
        groups[build_event_key(item)].append(item)

    session.query(EventItemLink).delete()
    session.query(EventTimelineEntry).delete()
    session.query(EventSummarySnapshot).delete()
    session.query(Event).delete()
    session.flush()

    for _, grouped_items in groups.items():
        grouped_items.sort(key=lambda item: item.event_time or item.created_at, reverse=True)
        latest = grouped_items[0]
        event = Event(
            title=latest.core_entity and f"{latest.core_entity} 相关事件" or latest.title,
            one_line_summary=build_one_line(grouped_items),
            primary_category_id=latest.category_id,
            status="active",
            heat_score=min(100, 60 + len(grouped_items) * 10),
            freshness_score=90,
            composite_score=min(100, 70 + len(grouped_items) * 8),
            source_count=len(grouped_items),
            started_at=grouped_items[-1].event_time or grouped_items[-1].created_at,
            last_updated_at=latest.event_time or latest.created_at,
        )
        session.add(event)
        session.flush()

        session.add_all(
            [
                EventSummarySnapshot(event_id=event.id, summary_type="one_line", content=event.one_line_summary, version=1),
                EventSummarySnapshot(event_id=event.id, summary_type="what_happened", content=latest.content[:180], version=1),
                EventSummarySnapshot(event_id=event.id, summary_type="why_it_matters", content=build_why_it_matters(grouped_items), version=1),
                EventSummarySnapshot(event_id=event.id, summary_type="latest_update", content=latest.content[:180], version=1),
            ]
        )

        for idx, item in enumerate(grouped_items, start=1):
            session.add(
                EventItemLink(
                    event_id=event.id,
                    item_id=item.id,
                    role="primary" if idx == 1 else "media",
                    is_primary=1 if idx == 1 else 0,
                    weight=max(10, 100 - idx * 10),
                )
            )
            session.add(
                EventTimelineEntry(
                    event_id=event.id,
                    occurred_at=item.event_time or item.created_at,
                    summary=item.content[:80],
                    source_item_id=item.id,
                    confidence=0.8,
                    display_order=idx,
                )
            )

    session.commit()

@app.get("/api/events")
def list_events(
    category_code: str = Query("all"),
    page: int = Query(1, ge=1),
    page_size: int = Query(10, ge=1, le=50),
):
    session = get_session()
    try:
        query = session.query(Event)
        if category_code != "all":
            query = query.join(Category, Category.id == Event.primary_category_id).filter(Category.code == category_code)

        total = query.count()
        items = (
            query.order_by(Event.composite_score.desc(), Event.last_updated_at.desc())
            .offset((page - 1) * page_size)
            .limit(page_size)
            .all()
        )
        return {
            "code": 0,
            "message": "success",

                "page": page,
                "page_size": page_size,

        event = session.query(Event).filter(Event.id == event_id).first()
        if not event:
            raise HTTPException(status_code=404, detail="事件不存在")

        summaries = {
            item.summary_type: item.content
            for item in session.query(EventSummarySnapshot).filter(EventSummarySnapshot.event_id == event_id).all()
        }
        timeline = (
            session.query(EventTimelineEntry)
            .filter(EventTimelineEntry.event_id == event_id)
            .order_by(EventTimelineEntry.occurred_at.asc())
            .all()
        )
        source_links = (
            session.query(EventItemLink, Info, Channel)
            .join(Info, Info.id == EventItemLink.item_id)
            .join(Channel, Channel.id == Info.channel_id)
            .filter(EventItemLink.event_id == event_id)
            .order_by(EventItemLink.weight.desc())
            .all()
        )

        return {
            "code": 0,
            "message": "success",
            "data": {
                "event": {
                    "id": event.id,
                    "title": event.title,
                    "one_line_summary": event.one_line_summary,
                    "primary_category": {"code": event.category.code, "name": event.category.name},
                    "heat_score": event.heat_score,
                    "last_updated_at": event.last_updated_at.strftime("%Y-%m-%d %H:%M:%S") if event.last_updated_at else None,
                },
                "timeline": [
                    {
                        "id": item.id,
                        "occurred_at": item.occurred_at.strftime("%Y-%m-%d %H:%M:%S"),
                        "summary": item.summary,
                        "confidence": item.confidence,
                    }
                    for item in timeline
                ],
                "summaries": summaries,
                "source_views": [
                    {"channel_name": channel.name, "summary": info.content[:120]}
                    for _, info, channel in source_links[:3]
                ],
                "representative_sources": [
                    {
                        "info_id": info.id,
                        "title": info.title,
                        "channel_name": channel.name,
                        "source_url": info.source_url,
                        "event_time": info.event_time.strftime("%Y-%m-%d %H:%M:%S") if info.event_time else None,
                    }
                    for _, info, channel in source_links[:6]
                ],
            },
        }
    finally:
        session.close()
